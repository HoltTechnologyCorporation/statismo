50 template <typename T>
51 UniquePtrType<typename ReducedVarianceModelBuilder<T>::StatisticalModelType>
52 ReducedVarianceModelBuilder<T>::BuildNewModelWithLeadingComponents(const StatisticalModelType * inputModel,
53  unsigned numberOfPrincipalComponents) const
54 
55 {
56  STATISMO_LOG_INFO("Building new model");
57  STATISMO_LOG_INFO("Number of principal components: " + std::to_string(numberOfPrincipalComponents));
58 
59  auto reducedModel =
60  StatisticalModelType::SafeCreate(inputModel->GetRepresenter(),
61  inputModel->GetMeanVector(),
62  inputModel->GetOrthonormalPCABasisMatrix().leftCols(numberOfPrincipalComponents),
63  inputModel->GetPCAVarianceVector().topRows(numberOfPrincipalComponents),
64  inputModel->GetNoiseVariance());
65 
66  // Write the parameters used to build the models into the builderInfo
67  typename ModelInfo::BuilderInfoList builderInfoList = inputModel->GetModelInfo().GetBuilderInfoList();
68 
69  BuilderInfo::ParameterInfoList bi;
70  bi.emplace_back("NumberOfPincipalComponents ", std::to_string(numberOfPrincipalComponents));
71 
72  BuilderInfo::DataInfoList di;
73  builderInfoList.emplace_back("ReducedVarianceModelBuilder", di, bi);
74 
75  // If the scores matrix is not set, or if we have for some reasons not as many score entries as the number of
76  // principal components, we simply work with what is there.
77  unsigned numComponentsForScores =
78  std::min(static_cast<unsigned>(inputModel->GetModelInfo().GetScoresMatrix().rows()), numberOfPrincipalComponents);
79 
80  reducedModel->SetModelInfo(
81  ModelInfo(inputModel->GetModelInfo().GetScoresMatrix().topRows(numComponentsForScores), builderInfoList));
82  return reducedModel;
83 }

86 template <typename T>
87 UniquePtrType<typename ReducedVarianceModelBuilder<T>::StatisticalModelType>
88 ReducedVarianceModelBuilder<T>::BuildNewModelWithVariance(const StatisticalModelType * inputModel,
89  double totalVariance) const
90 {
91  STATISMO_LOG_INFO("Building new model");
92  STATISMO_LOG_INFO("Total variance: " + std::to_string(totalVariance));
93 
94  VectorType pcaVariance = inputModel->GetPCAVarianceVector();
95  double modelVariance = pcaVariance.sum();
96 
97  // count the number of modes required for the model
98  double cumulatedVariance = 0;
99  unsigned numComponentsToReachPrescribedVariance = 0;
100  for (unsigned i = 0; i < pcaVariance.size(); i++)
101  {
102  cumulatedVariance += pcaVariance(i);
103  numComponentsToReachPrescribedVariance++;
104  if (cumulatedVariance / modelVariance >= totalVariance)
105  break;
106  }
107  return BuildNewModelWithLeadingComponents(inputModel, numComponentsToReachPrescribedVariance);
108 }
