167 VectorType
168 StatisticalModel<T>::DrawSampleVector(const VectorType & coefficients, bool addNoise) const
169 {
170  if (coefficients.size() != this->GetNumberOfPrincipalComponents())
171  {
172  STATISMO_LOG_DEBUG("Provided coefficient count: " + std::to_string(coefficients.size()));
173  STATISMO_LOG_DEBUG("Model component count: " + std::to_string(this->GetNumberOfPrincipalComponents()));
174  STATISMO_LOG_ERROR("Bad coefficient count");
175  throw StatisticalModelException("Incorrect number of coefficients provided");
176  }
177 
178  unsigned vectorSize = this->m_mean.size();
179  assert(vectorSize != 0);
180 
181  VectorType epsilon = VectorType::Zero(vectorSize);
182  if (addNoise)
183  {
184  epsilon = utils::GenerateNormalVector(vectorSize) * sqrt(m_noiseVariance);
185  }
186 
187 
188  return m_mean + m_pcaBasisMatrix * coefficients + epsilon;
189 }

201 template <typename T>
202 typename StatisticalModel<T>::ValueType
203 StatisticalModel<T>::DrawSampleAtPoint(const VectorType & coefficients, const unsigned ptId, bool addNoise) const
204 {
205  unsigned dim = m_representer->GetDimensions();
206 
207  VectorType v(dim);
208  VectorType epsilon = VectorType::Zero(dim);
209  if (addNoise)
210  {
211  epsilon = utils::GenerateNormalVector(dim) * sqrt(m_noiseVariance);
212  }
213  for (unsigned d = 0; d < dim; d++)
214  {
215  unsigned idx = m_representer->MapPointIdToInternalIdx(ptId, d);
216 
217  if (idx >= m_mean.rows())
218  {
219  STATISMO_LOG_DEBUG("Index: " + std::to_string(idx));
220  STATISMO_LOG_DEBUG("Mean rows count: " + std::to_string(m_mean.rows()));
221  STATISMO_LOG_ERROR("Invalid index");
222 
223  std::ostringstream os;
224  os << "Invalid idx computed in DrawSampleAtPoint.";
225  os << " The most likely cause of this error is that you provided an invalid point id (" << ptId << ")";
226  throw StatisticalModelException(os.str().c_str());
227  }
228 
229  v[d] = m_mean[idx] + m_pcaBasisMatrix.row(idx).dot(coefficients) + epsilon[d];
230  }
231 
232  return this->m_representer->PointSampleVectorToPointSample(v);
233 }

246 template <typename T>
247 MatrixType
248 StatisticalModel<T>::GetCovarianceAtPoint(unsigned ptId1, unsigned ptId2) const
249 {
250  unsigned dim = m_representer->GetDimensions();
251  MatrixType cov(dim, dim);
252 
253  for (unsigned i = 0; i < dim; i++)
254  {
255  unsigned idxi = m_representer->MapPointIdToInternalIdx(ptId1, i);
256  VectorType vi = m_pcaBasisMatrix.row(idxi);
257  for (unsigned j = 0; j < dim; j++)
258  {
259  unsigned idxj = m_representer->MapPointIdToInternalIdx(ptId2, j);
260  VectorType vj = m_pcaBasisMatrix.row(idxj);
261  cov(i, j) = vi.dot(vj);
262  if (i == j)
263  {
264  cov(i, j) += m_noiseVariance;
265  }
266  }
267  }
268  return cov;
269 }

317 template <typename T>
318 VectorType
319 StatisticalModel<T>::ComputeCoefficientsForPointIDValues(const PointIdValueListType & pointIdValueList,
320  double pointValueNoiseVariance) const
321 {
322 
323  unsigned dim = m_representer->GetDimensions();
324 
325  double noiseVariance = std::max(pointValueNoiseVariance, static_cast<double>(m_noiseVariance));
326 
327  // build the part matrices with , considering only the points that are fixed
328  MatrixType pcaBasisPart(pointIdValueList.size() * dim, this->GetNumberOfPrincipalComponents());
329  VectorType muPart(pointIdValueList.size() * dim);
330  VectorType sample(pointIdValueList.size() * dim);
331 
332  unsigned i{ 0 };
333  for (const auto & item : pointIdValueList)
334  {
335  VectorType val = this->m_representer->PointSampleToPointSampleVector(item.second);
336  unsigned ptId = item.first;
337  for (unsigned d = 0; d < dim; d++)
338  {
339  pcaBasisPart.row(i * dim + d) = this->GetPCABasisMatrix().row(m_representer->MapPointIdToInternalIdx(ptId, d));
340  muPart[i * dim + d] = this->GetMeanVector()[m_representer->MapPointIdToInternalIdx(ptId, d)];
341  sample[i * dim + d] = val[d];
342  }
343  i++;
344  }
345 
346  MatrixType matM = pcaBasisPart.transpose() * pcaBasisPart;
347  matM.diagonal() += noiseVariance * VectorType::Ones(pcaBasisPart.cols());
348  VectorType coeffs = matM.inverse() * pcaBasisPart.transpose() * (sample - muPart);
349 
350  return coeffs;
351 }

353 template <typename T>
354 VectorType
355 StatisticalModel<T>::ComputeCoefficientsForPointValuesWithCovariance(
356  const PointValueWithCovarianceListType & pointValuesWithCovariance) const
357 {
358 
359  // The naming of the variables correspond to those used in the paper
360  // Posterior Shape Models,
361  // Thomas Albrecht, Marcel Luethi, Thomas Gerig, Thomas Vetter
362  //
363  const MatrixType & matQ = m_pcaBasisMatrix;
364  const VectorType & mu = m_mean;
365 
366  unsigned dim = m_representer->GetDimensions();
367 
368  // build the part matrices with , considering only the points that are fixed
369  //
370  unsigned numPrincipalComponents = this->GetNumberOfPrincipalComponents();
371  MatrixType matQg(pointValuesWithCovariance.size() * dim, numPrincipalComponents);
372  VectorType mug(pointValuesWithCovariance.size() * dim);
373  VectorType sg(pointValuesWithCovariance.size() * dim);
374 
375  MatrixType matLQg(pointValuesWithCovariance.size() * dim, numPrincipalComponents);
376 
377  unsigned i = 0;
378  for (const auto & item : pointValuesWithCovariance)
379  {
380  VectorType val = m_representer->PointSampleToPointSampleVector(item.first.second);
381  unsigned ptId = m_representer->GetPointIdForPoint(item.first.first);
382 
383  // In the formulas, we actually need the precision matrix, which is the inverse of the covariance.
384  const MatrixType kPointPrecisionMatrix = item.second.inverse();
385 
386  // Get the three rows pertaining to this point:
387  const MatrixType kQrowsForPtId = matQ.block(ptId * dim, 0, dim, numPrincipalComponents);
388 
389  matQg.block(i * dim, 0, dim, numPrincipalComponents) = kQrowsForPtId;
390  mug.block(i * dim, 0, dim, 1) = mu.block(ptId * dim, 0, dim, 1);
391  sg.block(i * dim, 0, dim, 1) = val;
392 
393  matLQg.block(i * dim, 0, dim, numPrincipalComponents) = kPointPrecisionMatrix * kQrowsForPtId;
394  i++;
395  }
396 
397  VectorType d2 = m_pcaVariance.array();
398  const MatrixType & matQgT = matQg.transpose();
399 
400  MatrixType matM = matQgT * matLQg;
401  matM.diagonal() += VectorType::Ones(matQg.cols());
402 
403  MatrixTypeDoublePrecision matMinv = matM.cast<double>().inverse();
404 
405  // the MAP solution for the latent variables (coefficients)
406  VectorType coeffs = matMinv.cast<ScalarType>() * matLQg.transpose() * (sg - mug);
407 
408  return coeffs;
409 }

548 template <typename T>

550 StatisticalModel<T>::CheckAndUpdateCachedParameters() const
551 {
552 
553  if (!m_cachedValuesValid)
