102 template <class TPoint>
103 class GaussianKernel : public statismo::ScalarValuedKernel<TPoint>
104 {
105 public:
106  using CoordRepType = typename TPoint::CoordRepType;
107  using VectorType = vnl_vector<CoordRepType>;
108 
109  explicit GaussianKernel(double sigma)
110  : m_sigma(sigma)
111  , m_sigma2(-1.0 / (sigma * sigma))
112  {}
113 
114  inline double
115  operator()(const TPoint & x, const TPoint & y) const override
116  {
117  VectorType xv = x.GetVnlVector();
118  VectorType yv = y.GetVnlVector();
119 
120  VectorType r = yv - xv;
121 
122  return exp((double)dot_product(r, r) * m_sigma2);
123  }

128  std::ostringstream os;
129  os << "GaussianKernel(" << m_sigma << ")";
130  return os.str();
131  }

134  double m_sigma;
135  double m_sigma2;

139 template <class TPoint>
140 class BSplineKernel : public statismo::ScalarValuedKernel<TPoint>
141 {
142 public:
143  using CoordRepType = typename TPoint::CoordRepType;
144  using VectorType = vnl_vector<CoordRepType>;
145 
146  explicit BSplineKernel(double support)
147  : m_support(support)
148  {}
149 
150  double
151  bspline3(const double & x) const
152  {
153  const double absX = std::fabs(x);
154  const double absXSquared = absX * absX;
155  const double absXCube = absXSquared * absX;
156  const double twoMinAbsX = 2.0 - absX;
157  const double twoMinAbsXCube = twoMinAbsX * twoMinAbsX * twoMinAbsX;
158  const double twoByThree = 2.0 / 3.0;
159 
160  double splineValue = 0;
161  if (absX >= 0 && absX < 1)
162  {
163  splineValue = twoByThree - absXSquared + 0.5 * absXCube;
164  }
165  else if (absX >= 1 && absX < 2)
166  {
167  splineValue = twoMinAbsXCube / 6.0;
168  }
169  else
170  {
171  splineValue = 0;
172  }
173 
174  return splineValue;
175  }
176 
177  double
178  TensorProductSpline(const VectorType & x) const
179  {
180  double prod{ 1.0 };
181  for (unsigned d = 0; d < x.size(); ++d)
182  {
183  prod *= bspline3(x[d]);
184  }
185  return prod;
186  }
187 
188  inline double
189  operator()(const TPoint & x, const TPoint & y) const override
190  {
191  assert(x.Size() == y.Size());
192  const unsigned dim = x.Size();
193 
194  if (dim < 1 || dim > 3)
195  {
196  std::ostringstream os;
197  os << "Currently only dimensions 1 - 3 are supported. ( Received" << dim << ")";
198  throw statismo::StatisticalModelException(os.str().c_str(), statismo::Status::BAD_INPUT_ERROR);
199  }
200 
201  const double supportBasisFunction = 4.0;
202  const double scale = -1.0 * std::log(m_support / supportBasisFunction) / std::log(2.0);
203 
204  VectorType xScaled = x.GetVnlVector() * std::pow(2.0, scale);
205  VectorType yScaled = y.GetVnlVector() * std::pow(2.0, scale);
206 
207  std::vector<int> kLower(dim);
208  std::vector<int> kUpper(dim);
209  for (unsigned d = 0; d < dim; ++d)
210  {
211  kLower[d] = static_cast<int>(std::ceil(std::max(xScaled[d], yScaled[d]) - 0.5 * supportBasisFunction));
212  kUpper[d] = static_cast<int>(std::floor(std::min(xScaled[d], yScaled[d]) + 0.5 * supportBasisFunction));
213  }
214 
215  // We need to generate the cartesian product k_1 x ... x k_d, where k_i goes through all the integers
216  // within the given bounds. A non-recursive solution requires d loops. Here we just write down the cases
217  // for 1 2 and 3D
218  double sum = 0.0;
219  double kx = kLower[0];
220  while (kx <= kUpper[0])
221  {
222  if (dim == 1)
223  {
224  VectorType k(1);
225  k[0] = kx;
226  sum += (TensorProductSpline(xScaled - k) * TensorProductSpline(yScaled - k));
227  }
228  else
229  {
230  double ky = kLower[1];
231  while (ky <= kUpper[1])
232  {
233  if (dim == 2)
234  {
235  VectorType k(2);
236  k[0] = kx;
237  k[1] = ky;
238  sum += (TensorProductSpline(xScaled - k) * TensorProductSpline(yScaled - k));
239  }
240  else
241  {
242  double kz = kLower[2];
243  while (kz <= kUpper[2])
244  {
245  VectorType k(3);
246  k[0] = kx;
247  k[1] = ky;
248  k[2] = kz;
249  sum += (TensorProductSpline(xScaled - k) * TensorProductSpline(yScaled - k));
250  kz += 1;
251  }
252  }
253  ky += 1;
254  }
255  }
256 
257  kx += 1;
258  }
259 
260  return sum;
261  }

263  std::string
264  GetKernelInfo() const override
265  {
266  std::ostringstream os;
267  os << "BSplineKernel(" << m_support << ")";
268  return os.str();
269  }
270 
271 private:
272  double m_support;

275 //
276 // A kernel that represents deformations on different scale levels.
277 // Each scale level is modeled as a b-spline kernel, which leads to a "wavelet style"
278 // decomposition of the deformation. See
279 // Opfer, Roland. "Multiscale kernels." Advances in computational mathematics 25.4 (2006): 357-380.
280 //
281 template <class TPoint>
282 class MultiscaleKernel : public statismo::ScalarValuedKernel<TPoint>
283 {
284 public:
285  using CoordRepType = typename TPoint::CoordRepType;
286  using VectorType = vnl_vector<CoordRepType>;
287 
288  MultiscaleKernel(double supportBaseLevel, unsigned numberOfLevels)
289  : m_supportBaseLevel(supportBaseLevel)
290  , m_numberOfLevels(numberOfLevels)
291  {
292  for (unsigned i = 0; i < numberOfLevels; ++i)
293  {
294  m_kernels.push_back(std::make_unique<BSplineKernel<TPoint>>(m_supportBaseLevel * std::pow(2, -1.0 * i)));
295  m_kernelWeights.push_back(std::pow(2, -1.0 * i));
296  }
297  }
298 
299  inline double
300  operator()(const TPoint & x, const TPoint & y) const override
301  {
302  assert(x.Size() == y.Size());
303  double sum = 0;
304  for (unsigned i = 0; i < m_kernels.size(); ++i)
305  {
306  sum += (*m_kernels[i])(x, y) * m_kernelWeights[i];
307  }
308 
309  return sum;
310  }
311 
312  std::string
313  GetKernelInfo() const override
314  {
315  std::ostringstream os;
316  os << "MultiscaleKernel(" << m_supportBaseLevel << ", " << m_numberOfLevels << ")";
317  return os.str();
318  }
319 
320 private:
321  double m_supportBaseLevel;
322  unsigned m_numberOfLevels;
323  std::vector<std::unique_ptr<BSplineKernel<TPoint>>> m_kernels;
324  std::vector<double> m_kernelWeights;
325 };

334  static std::unique_ptr<statismo::ScalarValuedKernel<TPoint>>
335  CreateKernel(const std::vector<std::string> & kernelArgs)
336  {

352  }
353  else

376  itkGenericExceptionMacro(<< "Error: baselevel has to be > 0");

382  itkGenericExceptionMacro(<< "Error: could not parse the kernel argument");
