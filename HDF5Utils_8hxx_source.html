214 inline H5::Group
215 HDF5Utils::OpenPath(H5::H5File & file, const std::string & path, bool createPath)
216 {
217  H5::Group group;
218 
219  // take the first part of the path
220  std::size_t curpos = 1;
221  std::size_t nextpos = path.find_first_of('/', curpos);
222  auto g = file.openGroup("/");
223  auto name = path.substr(curpos, nextpos - 1);
224 
225  while (curpos != std::string::npos && !name.empty())
226  {
227  if (ExistsObjectWithName(g, name))
228  {
229  g = g.openGroup(name);
230  }
231  else
232  {
233  if (createPath)
234  {
235  g = g.createGroup(name);
236  }
237  else
238  {
239  std::string msg = std::string("the path ") + path + " does not exist";
240  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
241  }
242  }
243 
244  curpos = nextpos + 1;
245  nextpos = path.find_first_of('/', curpos);
246  if (nextpos != std::string::npos)
247  {
248  name = path.substr(curpos, nextpos - curpos);
249  }
250  else
251  {
252  name = path.substr(curpos);
253  }
254  }
255 
256  return g;
257 }

265 inline void
266 HDF5Utils::ReadMatrix(const H5::H5Location & fg, const char * name, unsigned maxNumColumns, MatrixType & matrix)
267 {
268  auto ds = fg.openDataSet(name);
269  hsize_t dims[2];
270  ds.getSpace().getSimpleExtentDims(dims, nullptr);
271 
272  auto nRows = dims[0]; // take the number of rows defined in the hdf5 file
273  auto nCols = std::min(dims[1], static_cast<hsize_t>(maxNumColumns)); // take the number of cols provided by the user
274 
275  hsize_t offset[2] = { 0, 0 }; // hyperslab offset in the file
276  hsize_t count[2];
277  count[0] = nRows;
278  count[1] = nCols;
279 
280  H5::DataSpace dataspace = ds.getSpace();
281  dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
282 
283  /* Define the memory dataspace. */
284  hsize_t dimsm[2];
285  dimsm[0] = nRows;
286  dimsm[1] = nCols;
287  H5::DataSpace memspace(2, dimsm);
288 
289  /* Define memory hyperslab. */
290  hsize_t offsetOut[2] = { 0, 0 }; // hyperslab offset in memory
291  hsize_t countOut[2]; // size of the hyperslab in memory
292 
293  countOut[0] = nRows;
294  countOut[1] = nCols;
295  memspace.selectHyperslab(H5S_SELECT_SET, countOut, offsetOut);
296 
297  matrix.resize(nRows, nCols);
298  ds.read(matrix.data(), H5::PredType::NATIVE_FLOAT, memspace, dataspace);
299 }
