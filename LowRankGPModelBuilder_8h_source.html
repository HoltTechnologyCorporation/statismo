136  UniquePtrType<StatisticalModelType>
137  BuildNewModel(typename RepresenterType::DatasetConstPointerType mean,
138  const MatrixValuedKernelType & kernel,
139  unsigned numComponents,
140  unsigned numPointsForNystrom = 500) const
141  {
142  STATISMO_LOG_INFO("Building new model");
143  STATISMO_LOG_INFO("Component count: " + std::to_string(numComponents));
144  auto domainPoints = m_representer->GetDomain().GetDomainPoints();
145  auto numDomainPoints = m_representer->GetDomain().GetNumberOfPoints();
146  auto kernelDim = kernel.GetDimension();
147 
148  auto nystrom = Nystrom<T>::SafeCreateStd(m_representer, kernel, numComponents, numPointsForNystrom);
149 
150  // We precompute the value of the eigenfunction for each domain point
151  // and store it later in the pcaBasis matrix. In this way we obtain
152  // a standard statismo model.
153  // To save time, we parallelize over the rows
154  unsigned numChunks = std::thread::hardware_concurrency() + 1;
155  ThreadPool pool{ numChunks - 1 };
156  std::vector<std::future<EigenfunctionComputationResult>> futvec;
157 
158  for (unsigned i = 0; i <= numChunks; i++)
159  {
160  unsigned chunkSize =
161  static_cast<unsigned>(ceil(static_cast<float>(numDomainPoints) / static_cast<float>(numChunks)));
162  unsigned lowerInd = i * chunkSize;
163  unsigned upperInd = std::min(static_cast<unsigned>(numDomainPoints), (i + 1) * chunkSize);
164 
165  if (lowerInd >= upperInd)
166  {
167  break;
168  }
169 
170  futvec.emplace_back(pool.Submit([=, nys = nystrom.get(), k = &kernel]() {
171  return ComputeEigenfunctionsForPoints(nys, k, numComponents, domainPoints, lowerInd, upperInd);
172  }));
173  }
174 
175  MatrixType pcaBasis = MatrixType::Zero(numDomainPoints * kernelDim, numComponents);
176 
177  // collect the result
178  for (auto & f : futvec)
179  {
180  auto res = f.get();
181  pcaBasis.block(res.lowerInd * kernelDim, 0, (res.upperInd - res.lowerInd) * kernelDim, pcaBasis.cols()) =
182  res.resultForPoints;
183  }
184  futvec.clear();
185 
186  STATISMO_LOG_DEBUG("End of multithreaded computation");
187 
188  auto pcaVariance = nystrom->GetEigenvalues();
189  auto mu = m_representer->SampleToSampleVector(mean);
190  auto model = StatisticalModelType::SafeCreate(m_representer, mu, pcaBasis, pcaVariance, 0);
191 
192  // the model builder does not use any data. Hence the scores and the datainfo is emtpy
193  MatrixType scores; // no scores
194  typename BuilderInfo::DataInfoList dataInfo;
195  typename BuilderInfo::ParameterInfoList bi;
196  bi.emplace_back(BuilderInfo::KeyValuePair("NoiseVariance", std::to_string(0)));
197  bi.emplace_back(BuilderInfo::KeyValuePair("KernelInfo", kernel.GetKernelInfo()));
198 
199  // finally add meta data to the model info
200  ModelInfo::BuilderInfoList biList(1, BuilderInfo{ "LowRankGPModelBuilder", dataInfo, bi });
201 
202  model->SetModelInfo(ModelInfo{ scores, biList });
203 
204  return model;
205  }
206 
207 private:
208  /*
209  * Compute the eigenfunction value at the poitns with index lowerInd - upperInd.
210  * Return a result object with the given values.
211  * This method is used to be able to parallelize the computations.
212  */
213  EigenfunctionComputationResult
214  ComputeEigenfunctionsForPoints(const Nystrom<T> * nystrom,
215  const MatrixValuedKernelType * kernel,
216  unsigned numEigenfunctions,
217  const std::vector<PointType> & domainPts,
218  unsigned lowerInd,
219  unsigned upperInd) const
220  {
221 
222  auto kernelDim = kernel->GetDimension();
223  assert(upperInd <= domainPts.size());
224 
225  // holds the results of the computation
226  MatrixType resMat = MatrixType::Zero((upperInd - lowerInd) * kernelDim, numEigenfunctions);
227 
228  // compute the nystrom extension for each point i in domainPts, for which
229  // i is in the right range
230  for (unsigned i = lowerInd; i < upperInd; i++)
231  {
232  auto pti = domainPts[i];
233  resMat.block((i - lowerInd) * kernelDim, 0, kernelDim, resMat.cols()) =
234  nystrom->ComputeEigenfunctionsAtPoint(pti);
235  }
236  return EigenfunctionComputationResult(lowerInd, upperInd, resMat);
237  }
