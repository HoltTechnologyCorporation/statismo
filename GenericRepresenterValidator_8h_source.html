88  TestSamplePointEvaluation() const
89  {
90  m_errString = "";
91  DatasetConstPointerType sample = m_testDataset;
92  auto id = m_representer->GetPointIdForPoint(m_testPoint);
93  auto val = m_representer->PointSampleFromSample(sample, id);
94  auto valVec = m_representer->PointSampleToPointSampleVector(val);
95 
96  // the obtained value should correspond to the value that is obtained by obtaining the sample vector, and evaluating
97  // it at the given position
98  auto sampleVector = m_representer->SampleToSampleVector(sample);
99  for (unsigned i = 0; i < m_representer->GetDimensions(); ++i)
100  {
101  if (sampleVector(i) != valVec(i))
102  {
103  return false;
104  }
105  }
106  return true;
107  }

112  bool
113  TestDomainValid() const
114  {
115  m_errString = "";
116  auto domain = m_representer->GetDomain();
117  auto domPoints = domain.GetDomainPoints();
118 
119  if (domPoints.size() == 0)
120  {
121  m_errString = "Representer defined on empty domain";
122  return false;
123  }
124 
125  if (domPoints.size() != domain.GetNumberOfPoints())
126  {
127  m_errString = "domPoints.size() != domain.GetNumberOfPoints() (" + std::to_string(domPoints.size()) +
128  " != " + std::to_string(domain.GetNumberOfPoints());
129  return false;
130  }
131  // if we convert a dataset to a samplevector, the resulting vector needs to have
132  // as many entries as there are points * dimensions
133  auto sampleVector = m_representer->SampleToSampleVector(m_testDataset);
134  if (sampleVector.rows() != static_cast<uint64_t>(m_representer->GetDimensions()) * domain.GetNumberOfPoints())
135  {
136  m_errString = "the dimension of the sampleVector does not agree with the number of points in the domain (#points "
137  "* dimensionality)";
138  return false;
139  }
140 
141  unsigned ptCounter{ 0 };
142  for (const auto & pt : domPoints)
143  {
144  // sample to every 10th points to improve performance
145  if (ptCounter % 10 != 0)
146  {
147  continue;
148  }
149 
150  if (m_representer->GetPointIdForPoint(pt) >= domain.GetNumberOfPoints())
151  {
152  m_errString = "a point in the domain did not evaluate to a valid point it";
153  return false;
154  }
155  ++ptCounter;
156  }
157 
158  return true;
159  }

203  bool
204  TestSampleVectorHasCorrectValueAtPoint() const
205  {
206  m_errString = "";
207  auto ptId = m_representer->GetPointIdForPoint(m_testPoint);
208  if (ptId < 0 || ptId >= m_representer->GetNumberOfPoints())
209  {
210  m_errString = "invalid point id for test point " + std::to_string(ptId);
211  return false;
212  }
213 
214  // the value of the point in the sample vector needs to correspond the the value that was provided
215  auto sampleVec = GetSampleVectorFromTestDataset();
216  auto pointSampleVec = m_representer->PointSampleToPointSampleVector(m_testValue);
217  for (unsigned d = 0; d < m_representer->GetDimensions(); ++d)
218  {
219  unsigned idx = m_representer->MapPointIdToInternalIdx(ptId, d);
220  if (sampleVec[idx] != pointSampleVec[d])
221  {
222  m_errString = "the sample vector does not contain the correct value of the pointSample";
223  return false;
224  }
225  }
226  return true;

232  bool
233  TestSaveLoad() const
234  {
235  m_errString = "";
236  using namespace H5;
237 
238  auto filename = statismo::utils::CreateTmpName(".rep");
239  H5File file;
240  try
241  {
242  file = H5File(filename, H5F_ACC_TRUNC);
243  }
244  catch (const Exception & e)
245  {
246  m_errString = "Error: Could not open HDF5 file for writing \n" + std::string(e.getCDetailMsg());
247  return false;
248  }
249  auto representerGroup = file.createGroup("/representer");
250  m_representer->Save(representerGroup);
251 
252  // We add the required attributes, which are usually written by the StatisticalModel class.
253  // This is needed, as some representers check on these values.
254  statismo::HDF5Utils::WriteStringAttribute(representerGroup, "name", m_representer->GetName());
255  std::string dataTypeStr = Representer::TypeToString(m_representer->GetType());
256  statismo::HDF5Utils::WriteStringAttribute(representerGroup, "datasetType", dataTypeStr);
257 
258  file.close();
259  try
260  {
261  file = H5File(filename.c_str(), H5F_ACC_RDONLY);
262  }
263  catch (const Exception & e)
264  {
265  m_errString = "Error: could not open HDF5 file \n" + std::string(e.getCDetailMsg());
266  return false;
267  }
268 
269  representerGroup.close();
270  representerGroup = file.openGroup("/representer");
271 
272  auto newRep = Representer::SafeCreate();
273  newRep->Load(representerGroup);
274 
275  return AssertRepresenterEqual(newRep.get(), m_representer);
276  }
