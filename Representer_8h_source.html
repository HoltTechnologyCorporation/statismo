423  DatasetPointerType

425  {
426 
427  switch (this->GetType())
428  {
429  case RepresenterDataType::POINT_SET:
430  case RepresenterDataType::POLYGON_MESH:
431  case RepresenterDataType::VOLUME_MESH:
432  {
433  return this->CloneDataset(this->GetReference());
434  break;
435  }
436  case RepresenterDataType::IMAGE:
437  case RepresenterDataType::VECTOR:
438  {
439  VectorType zeroVec =
440  VectorType::Zero(static_cast<uint64_t>(this->GetDomain().GetNumberOfPoints()) * GetDimensions());
441  return this->SampleVectorToSample(zeroVec);
442  break;
443  }
444  default:
445  {
446  throw statismo::StatisticalModelException(
447  "No cannonical identityDataset method is defined for custom Representers.");
448  }
449  }
450  }
