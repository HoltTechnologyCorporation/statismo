437  DatasetPointerType

440 
441  switch (this->GetType())

443  case RepresenterDataType::POINT_SET:
444  case RepresenterDataType::POLYGON_MESH:
445  case RepresenterDataType::VOLUME_MESH:

453  VectorType zeroVec =
454  VectorType::Zero(static_cast<uint64_t>(this->GetDomain().GetNumberOfPoints()) * GetDimensions());
455  return this->SampleVectorToSample(zeroVec);
456  break;
457  }
458  default:

461  "No cannonical identityDataset method is defined for custom Representers.");
462  }
463  }
