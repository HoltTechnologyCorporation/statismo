106 class WorkStealingQueue : public NonCopyable
107 {
108 public:
109  using TaskType = FunctionWrapper;
110 
111  void
112  Push(TaskType data)
113  {
114  std::lock_guard<std::mutex> lock{ m_mut };
115  m_queue.push_front(std::move(data));
116  }
117 
118  bool
119  Empty() const
120  {
121  std::lock_guard<std::mutex> lock{ m_mut };
122  return m_queue.empty();
123  }
124 
125  bool
126  TryPop(TaskType & t)
127  {
128  std::lock_guard<std::mutex> lock{ m_mut };
129 
130  if (m_queue.empty())
131  {
132  return false;
133  }
134 
135  t = std::move(m_queue.front());
136  m_queue.pop_front();
137  return true;
138  }
139 
140  bool
141  TrySteal(TaskType & t)
142  {
143  std::lock_guard<std::mutex> lock{ m_mut };
144 
145  if (m_queue.empty())
146  {
147  return false;
148  }
149 
150  t = std::move(m_queue.back());
151  m_queue.pop_back();
152  return true;
153  }
154 
155 private:
156  std::deque<TaskType> m_queue;
157  mutable std::mutex m_mut;
158 };

227 class ThreadPool final : public NonCopyable
228 {
229 public:
230  enum class WaitingMode
231  {
232  YIELD = 0,
233  WAIT_FOR = 1
234  };
235 
236  using TaskType = FunctionWrapper;
237 
238  explicit ThreadPool(unsigned maxThreads, WaitingMode m, unsigned waitTime)
239  : m_waitMode{ m }
240  , m_waitTime{ waitTime }
241  {
242  const auto kThreadCount = std::min(std::thread::hardware_concurrency(), maxThreads);
243 
244  for (std::remove_cv_t<decltype(kThreadCount)> i = 0; i < kThreadCount; ++i)
245  {
246  m_queues.push_back(std::make_unique<WorkStealingQueue>());
247  }
248 
249  try
250  {
251  for (std::remove_cv_t<decltype(kThreadCount)> i = 0; i < kThreadCount; ++i)
252  {
253  m_threads.push_back(RaiiThread{ std::thread{ &ThreadPool::DoThreadJob, this, i } });
254  }
255  }
256  catch (...)
257  {
258  m_isDone = true;
259  throw StatisticalModelException("Failed to create thread pool");
260  }
261  }
262 
263  explicit ThreadPool(unsigned maxThreads = std::numeric_limits<unsigned int>::max())
264  : ThreadPool{ maxThreads, WaitingMode::YIELD, 0 }
265  {}
266 
267  virtual ~ThreadPool() { m_isDone = true; } // NOLINT
268 
269  template <typename F>
270  std::future<std::invoke_result_t<F>>
271  Submit(F t)
272  {
273  std::packaged_task<std::invoke_result_t<F>()> task{ t };
274  std::future<std::invoke_result_t<F>> res{ task.get_future() };
275 
276  if (m_localQueue != nullptr)
277  {
278  m_localQueue->Push(std::move(task));
279  }
280  else
281  {
282  m_poolQueue.Push(std::move(task));
283  }
284 
285  return res;
286  }
287 
288 private:
289  void
290  DoThreadJob(std::size_t idx)
291  {
292  m_tid = idx;
293  m_localQueue = m_queues[m_tid].get();
294 
295  while (!m_isDone)
296  {
297  RunPendingTask();
298  }
299  }
300 
301  bool
302  PopTaskFromLocalQueue(TaskType & t)
303  {
304  return m_localQueue->TryPop(t);
305  }
306 
307  bool
308  PopTaskFromPoolQueue(TaskType & t)
309  {
310  return m_poolQueue.TryPop(t);
311  }
312 
313  bool
314  PopTaskFromOtherLocalQueues(TaskType & t)
315  {
316  for (std::size_t i = 0; i < m_queues.size(); ++i)
317  {
318  if (m_queues[(m_tid + i + 1) % m_queues.size()]->TrySteal(t))
319  {
320  return true;
321  }
322  }
323 
324  return false;
325  }
326 
327  void
328  RunPendingTask()
329  {
330  TaskType t;
331 
332  if (PopTaskFromLocalQueue(t) || PopTaskFromPoolQueue(t) || PopTaskFromOtherLocalQueues(t))
333  {
334  t();
335  }
336  else
337  {
338  if (m_waitMode == WaitingMode::YIELD)
339  {
340  std::this_thread::yield();
341  }
342  else
343  {
344  std::this_thread::sleep_for(std::chrono::milliseconds(m_waitTime));
345  }
346  }
347  }
348 
349  std::atomic_bool m_isDone{ false };
350  WaitingMode m_waitMode{ WaitingMode::YIELD };
351  unsigned m_waitTime{ 0 };
352  SafeQueue<TaskType> m_poolQueue;
353  std::vector<std::unique_ptr<WorkStealingQueue>> m_queues;
354  std::vector<RaiiThread> m_threads;
355 
356  static thread_local WorkStealingQueue * m_localQueue;
357  static thread_local std::size_t m_tid;
358 };
