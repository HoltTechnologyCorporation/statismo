63 UniquePtrType<DataManagerBase<T, Derived>>
64 DataManagerBase<T, Derived>::Load(RepresenterType * representer, const std::string & filename, Args &&... args)
65 {
66  using namespace H5;
67 
68  H5File file;
69  try
70  {
71  file = H5File(filename.c_str(), H5F_ACC_RDONLY);
72  }
73  catch (const H5::Exception & e)
74  {
75  std::string msg(std::string("could not open HDF5 file \n") + e.getCDetailMsg());
76  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
77  }
78 
79  UniquePtrType<DataManagerBase> newDataManagerBase;
80  try
81  {
82  // loading representer
83  auto representerGroup = file.openGroup("./representer");
84  auto repName = HDF5Utils::ReadStringAttribute(representerGroup, "name");
85  auto repTypeStr = HDF5Utils::ReadStringAttribute(representerGroup, "datasetType");
86  auto versionStr = HDF5Utils::ReadStringAttribute(representerGroup, "version");
87  auto type = RepresenterType::TypeFromString(repTypeStr);
88  if (type == RepresenterDataType::CUSTOM || type == RepresenterDataType::UNKNOWN)
89  {
90  if (repName != representer->GetName())
91  {
92  std::ostringstream os;
93  os << "A different representer was used to create the file and the representer is not of a standard type ";
94  os << ("(RepresenterName = ") << repName << " does not match required name = " << representer->GetName() << ")";
95  os << "Cannot load hdf5 file";
96  throw StatisticalModelException(os.str().c_str(), Status::INVALID_DATA_ERROR);
97  }
98 
99  if (versionStr != representer->GetVersion())
100  {
101  std::ostringstream os;
102  os << "The version of the representers do not match ";
103  os << ("(Version = ") << versionStr << " != = " << representer->GetVersion() << ")";
104  os << "Cannot load hdf5 file";
105  }
106  }
107 
108  if (type != representer->GetType())
109  {
110  std::ostringstream os;
111  os << "The representer that was provided cannot be used to load the dataset ";
112  os << "(" << RepresenterType::TypeToString(type)
113  << " != " << RepresenterType::TypeToString(representer->GetType()) << ").";
114  os << "Cannot load hdf5 file.";
115  throw StatisticalModelException(os.str().c_str(), Status::INVALID_DATA_ERROR);
116  }
117 
118  representer->Load(representerGroup);
119  newDataManagerBase =
120  UniquePtrType<DataManagerBase>(DataManagerBase::Create(representer, std::forward<Args>(args)...));
121 
122  auto dataGroup = file.openGroup("/data");
123  unsigned numds = HDF5Utils::ReadInt(dataGroup, "./NumberOfDatasets");
124  for (unsigned num = 0; num < numds; num++)
125  {
126  std::ostringstream ss;
127  ss << "./dataset-" << num;
128 
129  auto dsGroup = file.openGroup(ss.str().c_str());
130  newDataManagerBase->m_dataItemList.push_back(ConcreteDataItemType::Load(representer, dsGroup));
131  }
132  }
133  catch (const H5::Exception & e)
134  {
135  std::string msg =
136  std::string("an exception occurred while reading data matrix to HDF5 file \n") + e.getCDetailMsg();
137  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
138  }
139 
140  assert(newDataManagerBase);
141  return newDataManagerBase;
142 }

144 template <typename T, typename Derived>
145 void
146 DataManagerBase<T, Derived>::Save(const std::string & filename) const
147 {
148  STATISMO_LOG_INFO("Saving data to file " + filename);
149 
150  using namespace H5;
151 
152  assert(m_representer);
153 
154  H5File file;
155  try
156  {
157  file = H5File(filename.c_str(), H5F_ACC_TRUNC);
158  }
159  catch (const H5::Exception & e)
160  {
161  std::string msg(std::string("Could not open HDF5 file for writing \n") + e.getCDetailMsg());
162  STATISMO_LOG_ERROR(msg);
163  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
164  }
165 
166  try
167  {
168  auto representerGroup = file.createGroup("./representer");
169  auto dataTypeStr = RepresenterType::TypeToString(m_representer->GetType());
170 
171  HDF5Utils::WriteStringAttribute(representerGroup, "name", m_representer->GetName());
172  HDF5Utils::WriteStringAttribute(representerGroup, "version", m_representer->GetVersion());
173  HDF5Utils::WriteStringAttribute(representerGroup, "datasetType", dataTypeStr);
174 
175  this->m_representer->Save(representerGroup);
176 
177  auto dataGroup = file.createGroup("./data");
178 
179  if (this->m_dataItemList.size() > std::numeric_limits<int>::max())
180  {
181  STATISMO_LOG_ERROR("Too many dataset to write");
182  throw StatisticalModelException("too many dataset to write", Status::OUT_OF_RANGE_ERROR);
183  }
184 
185  HDF5Utils::WriteInt(dataGroup, "./NumberOfDatasets", static_cast<int>(this->m_dataItemList.size()));
186 
187  unsigned num{ 0 };
188  for (const auto & item : m_dataItemList)
189  {
190  std::ostringstream ss;
191  ss << "./dataset-" << num;
192 
193  Group dsGroup = file.createGroup(ss.str().c_str());
194  item->Save(dsGroup);
195 
196  num++;
197  }
198  }
199  catch (const H5::Exception & e)
200  {
201  std::string msg =
202  std::string("an exception occurred while writing data matrix to HDF5 file \n") + e.getCDetailMsg();
203  STATISMO_LOG_ERROR(msg);
204  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
205  }
206 }

215 template <typename T, typename Derived>
216 typename DataManagerBase<T, Derived>::CrossValidationFoldListType
217 DataManagerBase<T, Derived>::GetCrossValidationFolds(unsigned nFolds, bool isRandomized) const
218 {
219  STATISMO_LOG_INFO("Retrieving validation folds");
220  STATISMO_LOG_INFO("Folds count: " + std::to_string(nFolds));
221  STATISMO_LOG_INFO("Sample count: " + std::to_string(GetNumberOfSamples()));
222 
223  if (nFolds <= 1 || nFolds > GetNumberOfSamples())
224  {
225  throw StatisticalModelException("Invalid number of folds specified in GetCrossValidationFolds",
226  Status::BAD_INPUT_ERROR);
227  }
228 
229  auto nElemsPerFold = GetNumberOfSamples() / nFolds;
230 
231  // we create a vector with as many entries as datasets. Each entry contains the
232  // fold the entry belongs to
233  std::vector<std::size_t> batchAssignment(GetNumberOfSamples());
234 
235  for (std::size_t i = 0; i < GetNumberOfSamples(); i++)
236  {
237  batchAssignment[i] = std::min(i / nElemsPerFold, static_cast<std::size_t>(nFolds));
238  }
239 
240  // randomly shuffle the vector
241  if (isRandomized)
242  {
243  std::random_device rd;
244  std::mt19937 g(rd());
245  std::shuffle(std::begin(batchAssignment), std::end(batchAssignment), g);
246  }
247 
248  // now we create the folds
249  CrossValidationFoldListType foldList;
250  for (unsigned currentFold = 0; currentFold < nFolds; currentFold++)
251  {
252  DataItemListType trainingData;
253  DataItemListType testingData;
254 
255  unsigned sampleNum{ 0 };
256  for (const auto & item : m_dataItemList)
257  {
258  if (batchAssignment[sampleNum] != currentFold)
259  {
260  trainingData.push_back(item);
261  }
262  else
263  {
264  testingData.push_back(item);
265  }
266  ++sampleNum;
267  }
268  foldList.emplace_back(trainingData, testingData);
269  }
270  return foldList;
271 }

273 template <typename T, typename Derived>
274 typename DataManagerBase<T, Derived>::CrossValidationFoldListType
275 DataManagerBase<T, Derived>::GetLeaveOneOutCrossValidationFolds() const
276 {
277  STATISMO_LOG_INFO("Retrieving validation folds");
278  CrossValidationFoldListType foldList;
279  for (unsigned currentFold = 0; currentFold < GetNumberOfSamples(); currentFold++)
280  {
281  DataItemListType trainingData;
282  DataItemListType testingData;
283 
284  unsigned sampleNum{ 0 };
285  for (const auto & item : m_dataItemList)
286  {
287  if (sampleNum == currentFold)
288  {
289  testingData.push_back(item);
290  }
291  else
292  {
293  trainingData.push_back(item);
294  }
295  ++sampleNum;
296  }
297  CrossValidationFoldType fold(trainingData, testingData);
298  foldList.push_back(fold);
299  }
300  return foldList;
301 }
