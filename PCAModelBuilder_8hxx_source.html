51 template <typename T>
52 UniquePtrType<typename PCAModelBuilder<T>::StatisticalModelType>
53 PCAModelBuilder<T>::BuildNewModel(const DataItemListType & sampleDataList,
54  double noiseVariance,
55  bool computeScores,
56  EigenValueMethod method) const
57 {
58 
59  STATISMO_LOG_INFO("Building new model");
60  STATISMO_LOG_INFO("Noise variance: " + std::to_string(noiseVariance));
61 
62  auto n = sampleDataList.size();
63  STATISMO_LOG_INFO("Sample count: " + std::to_string(n));
64 
65  if (n <= 0)
66  {
67  throw StatisticalModelException("Provided empty sample set. Cannot build the sample matrix",
68  Status::BAD_INPUT_ERROR);
69  }
70 
71  unsigned p = sampleDataList.front()->GetSampleVector().rows();
72  const auto * representer = sampleDataList.front()->GetRepresenter();
73 
74  // Compute the mean vector mu
75  VectorType mu = VectorType::Zero(p);
76 
77  for (const auto & item : sampleDataList)
78  {
79  assert(item->GetSampleVector().rows() == p); // all samples must have same number of rows
80  assert(item->GetRepresenter() == representer); // all samples have the same representer
81  mu += item->GetSampleVector();
82  }
83 
84  mu /= n;
85 
86  // Build the mean free sample matrix X0
87  MatrixType matX0(n, p);
88  unsigned i{ 0 };
89  for (const auto & item : sampleDataList)
90  {
91  matX0.row(i++) = item->GetSampleVector() - mu;
92  }
93 
94  // build the model
95  auto model = BuildNewModelInternal(representer, matX0, mu, noiseVariance, method);
96 
97  // compute the scores if requested
98  MatrixType scores;
99  if (computeScores)
100  {
101  scores = this->ComputeScores(sampleDataList, model.get());
102  }
103 
104  typename BuilderInfo::ParameterInfoList bi;
105  bi.emplace_back(BuilderInfo::KeyValuePair("NoiseVariance ", std::to_string(noiseVariance)));
106 
107  typename BuilderInfo::DataInfoList dataInfo;
108  i = 0;
109  for (const auto & item : sampleDataList)
110  {
111  STATISMO_LOG_INFO("Adding info about dataset with uri " + item->GetDatasetURI());
112  std::ostringstream os;
113  os << "URI_" << i++;
114  dataInfo.emplace_back(os.str().c_str(), item->GetDatasetURI());
115  }
116 
117  // finally add meta data to the model info
118  ModelInfo::BuilderInfoList biList;
119  biList.emplace_back("PCAModelBuilder", dataInfo, bi);
120 
121  model->SetModelInfo(ModelInfo{ scores, biList });
122 
123  return model;
124 }

127 template <typename T>
128 UniquePtrType<typename PCAModelBuilder<T>::StatisticalModelType>
129 PCAModelBuilder<T>::BuildNewModelInternal(const Representer<T> * representer,
130  const MatrixType & matX0,
131  const VectorType & mu,
132  double noiseVariance,
133  EigenValueMethod method) const
134 {
135  unsigned n = matX0.rows();
136  unsigned p = matX0.cols();
137 
138  STATISMO_LOG_INFO("n: " + std::to_string(n));
139  STATISMO_LOG_INFO("p: " + std::to_string(p));
140 
141  switch (method)
142  {
143  case EigenValueMethod::JACOBI_SVD:
144 
145  STATISMO_LOG_INFO("Using JACOBI_SVD method");
146 
147  using SVDType = Eigen::JacobiSVD<MatrixType>;
148  using SVDDoublePrecisionType = Eigen::JacobiSVD<MatrixTypeDoublePrecision>;
149 
150  // We destinguish the case where we have more variables than samples and
151  // the case where we have more samples than variable.
152  // In the first case we compute the (smaller) inner product matrix instead of the full covariance matrix.
153  // It is known that this has the same non-zero singular values as the covariance matrix.
154  // Furthermore, it is possible to compute the corresponding eigenvectors of the covariance matrix from the
155  // decomposition.
156 
157  if (n < p)
158  {
159  // we compute the eigenvectors of the covariance matrix by computing an SVD of the
160  // n x n inner product matrix 1/(n-1) X0X0^T
161  MatrixType cov = matX0 * matX0.transpose() * 1.0 / (n - 1);
162  SVDDoublePrecisionType svd(cov.cast<double>(), Eigen::ComputeThinV);
163  VectorType singularValues = svd.singularValues().cast<ScalarType>();
164  MatrixType matV = svd.matrixV().cast<ScalarType>();
165 
166  unsigned numComponentsAboveTolerance =
167  ((singularValues.array() - noiseVariance - Superclass::sk_tolerance) > 0).count();
168 
169  // there can be at most n-1 nonzero singular values in this case. Everything else must be due to numerical
170  // inaccuracies
171  unsigned numComponentsToKeep = std::min(numComponentsAboveTolerance, n - 1);
172  // compute the pseudo inverse of the square root of the singular values
173  // which is then needed to recompute the PCA basis
174  VectorType singSqrt = singularValues.array().sqrt();
175  VectorType singSqrtInv = VectorType::Zero(singSqrt.rows());
176  for (unsigned i = 0; i < numComponentsToKeep; i++)
177  {
178  assert(singSqrt(i) > Superclass::sk_tolerance);
179  singSqrtInv(i) = 1.0 / singSqrt(i);
180  }
181 
182  if (numComponentsToKeep == 0)
183  {
184  STATISMO_LOG_ERROR("No component to keep");
185  throw StatisticalModelException("All the eigenvalues are below the given tolerance. Model cannot be built.");
186  }
187 
188  // we recover the eigenvectors U of the full covariance matrix from the eigenvectors V of the inner product
189  // matrix. We use the fact that if we decompose X as X=UDV^T, then we get X^TX = UD^2U^T and XX^T = VD^2V^T
190  // (exploiting the orthogonormality of the matrix U and V from the SVD). The additional factor sqrt(n-1) is to
191  // compensate for the 1/sqrt(n-1) in the formula for the covariance matrix.
192 
193  MatrixType pcaBasis = matX0.transpose() * matV * singSqrtInv.asDiagonal();
194  pcaBasis /= sqrt(n - 1.0);
195  pcaBasis.conservativeResize(Eigen::NoChange, numComponentsToKeep);
196 
197 
198  VectorType sampleVarianceVector = singularValues.topRows(numComponentsToKeep);
199  VectorType pcaVariance = (sampleVarianceVector - VectorType::Ones(numComponentsToKeep) * noiseVariance);
200 
201  auto model = StatisticalModelType::SafeCreate(representer, mu, pcaBasis, pcaVariance, noiseVariance);
202 
203  return model;
204  }
205  else // NOLINT
206  {
207  // we compute an SVD of the full p x p covariance matrix 1/(n-1) X0^TX0 directly
208  SVDType svd(matX0.transpose() * matX0, Eigen::ComputeThinU);
209  VectorType singularValues = svd.singularValues();
210  singularValues /= (n - 1.0);
211  unsigned numComponentsToKeep =
212  ((singularValues.array() - noiseVariance - Superclass::sk_tolerance) > 0).count();
213  MatrixType pcaBasis = svd.matrixU();
214 
215  pcaBasis.conservativeResize(Eigen::NoChange, numComponentsToKeep);
216 
217  if (numComponentsToKeep == 0)
218  {
219  STATISMO_LOG_ERROR("No component to keep");
220  throw StatisticalModelException("All the eigenvalues are below the given tolerance. Model cannot be built.");
221  }
222 
223  VectorType sampleVarianceVector = singularValues.topRows(numComponentsToKeep);
224  VectorType pcaVariance = (sampleVarianceVector - VectorType::Ones(numComponentsToKeep) * noiseVariance);
225  auto model = StatisticalModelType::SafeCreate(representer, mu, pcaBasis, pcaVariance, noiseVariance);
226  return model;
227  }
228  break;
229 
230  case EigenValueMethod::SELF_ADJOINT_EIGEN_SOLVER:
231  {
232  STATISMO_LOG_INFO("Using SELF_ADJOINT_EIGEN_SOLVER method");
233  // we compute the eigenvalues/eigenvectors of the full p x p covariance matrix 1/(n-1) X0^TX0 directly
234 
235  using SelfAdjointEigenSolver = Eigen::SelfAdjointEigenSolver<MatrixType>;
236  SelfAdjointEigenSolver es;
237  es.compute(matX0.transpose() * matX0);
238  VectorType eigenValues =
239  es.eigenvalues().reverse(); // SelfAdjointEigenSolver orders the eigenvalues in increasing order
240  eigenValues /= (n - 1.0);
241 
242  unsigned numComponentsToKeep = ((eigenValues.array() - noiseVariance - Superclass::sk_tolerance) > 0).count();
243  MatrixType pcaBasis = es.eigenvectors().rowwise().reverse();
244  pcaBasis.conservativeResize(Eigen::NoChange, numComponentsToKeep);
245 
246  if (numComponentsToKeep == 0)
247  {
248  throw StatisticalModelException("All the eigenvalues are below the given tolerance. Model cannot be built.");
249  }
250 
251  VectorType sampleVarianceVector = eigenValues.topRows(numComponentsToKeep);
252  VectorType pcaVariance = (sampleVarianceVector - VectorType::Ones(numComponentsToKeep) * noiseVariance);
253  auto model = StatisticalModelType::SafeCreate(representer, mu, pcaBasis, pcaVariance, noiseVariance);
254  return model;
255  }
256  break;
257 
258  default:
259  throw StatisticalModelException("Unrecognized decomposition/eigenvalue solver method.");
260  }
261  return nullptr;
262 }
