93 template <typename Pixel, unsigned IMAGE_DIMENSION>
94 typename StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::ImageType::Pointer
95 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::LoadRef(const H5::Group & fg) const
96 {
97  auto readImageDimension = statismo::HDF5Utils::ReadInt(fg, "imageDimension");
98  if (readImageDimension != IMAGE_DIMENSION)
99  {
100  throw statismo::StatisticalModelException(
101  "the image dimension specified in the statismo file does not match the one specified as template parameter",
102  statismo::Status::INVALID_H5DATA_ERROR);
103  }
104 
105  // Read
106  statismo::VectorType originVec;
107  statismo::HDF5Utils::ReadVector(fg, "origin", originVec);
108 
109  statismo::VectorType spacingVec;
110  statismo::HDF5Utils::ReadVector(fg, "spacing", spacingVec);
111 
112  typename statismo::GenericEigenTraits<int>::VectorType sizeVec;
113  statismo::HDF5Utils::ReadVectorOfType<int>(fg, "size", sizeVec);
114 
115  typename ImageType::PointType origin;
116  typename ImageType::SpacingType spacing;
117  typename ImageType::SizeType size;
118  for (unsigned i = 0; i < IMAGE_DIMENSION; ++i)
119  {
120  origin[i] = originVec[i];
121  spacing[i] = spacingVec[i];
122  size[i] = sizeVec[i];
123  }
124 
125  statismo::MatrixType directionMat;
126  statismo::HDF5Utils::ReadMatrix(fg, "direction", directionMat);
127  typename ImageType::DirectionType direction;
128  for (unsigned i = 0; i < directionMat.rows(); ++i)
129  {
130  for (unsigned j = 0; j < directionMat.rows(); ++j)
131  {
132  direction[i][j] = directionMat(i, j);
133  }
134  }
135 
136  H5::Group pdGroup = fg.openGroup("./pointData");
137  auto readPixelDimension = static_cast<unsigned>(statismo::HDF5Utils::ReadInt(pdGroup, "pixelDimension"));
138  if (readPixelDimension != StandardImageRepresenter::GetDimensions())
139  {
140  throw statismo::StatisticalModelException(
141  "the pixel dimension specified in the statismo file does not match the one specified as template parameter",
142  statismo::Status::INVALID_H5DATA_ERROR);
143  }
144 
145  typename statismo::GenericEigenTraits<double>::MatrixType pixelMatDouble;
146  statismo::HDF5Utils::ReadMatrixOfType<double>(pdGroup, "pixelValues", pixelMatDouble);
147  statismo::MatrixType pixelMat = pixelMatDouble.cast<statismo::ScalarType>();
148  typename ImageType::Pointer newImage = ImageType::New();
149  typename ImageType::IndexType start;
150  start.Fill(0);
151 
152  auto ds = pdGroup.openDataSet("pixelValues");
153  auto type = static_cast<unsigned>(statismo::HDF5Utils::ReadIntAttribute(ds, "datatype"));
154  if (type != PixelConversionTrait<Pixel>::GetDataType())
155  {
156  STATISMO_LOG_WARNING(
157  "The datatype specified for the scalars does not match the Pixel template argument used in this representer");
158  }
159 
160  typename ImageType::RegionType region(start, size);
161  newImage->SetRegions(region);
162  newImage->Allocate();
163  newImage->SetOrigin(origin);
164  newImage->SetSpacing(spacing);
165  newImage->SetDirection(direction);
166 
167  itk::ImageRegionIterator<DatasetType> it(newImage, newImage->GetLargestPossibleRegion());
168  it.GoToBegin();
169  for (unsigned i = 0; !(it.IsAtEnd()); ++it, ++i)
170  {
171  Pixel v = PixelConversionTrait<Pixel>::FromVector(pixelMat.col(i));
172  it.Set(v);
173  }
174 
175  return newImage;
176 }

178 template <typename Pixel, unsigned IMAGE_DIMENSION>
179 typename StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::ImageType::Pointer
180 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::LoadRefLegacy(const H5::Group & fg) const
181 {
182  auto tmpfilename = statismo::utils::CreateTmpName(".vtk");
183  statismo::HDF5Utils::GetFileFromHDF5(fg, "./reference", tmpfilename.c_str());
184 
185  auto uw = statismo::MakeStackUnwinder([=]() { statismo::utils::RemoveFile(tmpfilename); });
186 
187  auto reader = itk::ImageFileReader<ImageType>::New();
188  reader->SetFileName(tmpfilename);
189  try
190  {
191  reader->Update();
192  }
193  catch (const itk::ImageFileReaderException & e)
194  {
195  throw statismo::StatisticalModelException((std::string("Could not read file ") + tmpfilename).c_str(),
196  statismo::Status::IO_ERROR);
197  }
198  typename DatasetType::Pointer img = reader->GetOutput();
199  img->Register();
200  return img;
201 }

205 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::SetReference(ImageType * reference)
206 {
207  m_reference = reference;
208 
209  typename DomainType::DomainPointsListType domainPoints;
210  itk::ImageRegionConstIterator<DatasetType> it(reference, reference->GetLargestPossibleRegion());
211  for (it.GoToBegin(); !(it.IsAtEnd()); ++it)
212  {
213  PointType pt;

215  domainPoints.push_back(pt);
216  }

232 template <typename Pixel, unsigned IMAGE_DIMENSION>
233 statismo::VectorType
234 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::SampleToSampleVector(DatasetConstPointerType sample) const
235 {
236  statismo::VectorType sampleVec(this->GetNumberOfPoints() * StandardImageRepresenter::GetDimensions());
237  itk::ImageRegionConstIterator<DatasetType> it(sample, sample->GetLargestPossibleRegion());
238 
239  it.GoToBegin();
240  for (unsigned i = 0; !(it.IsAtEnd()); ++i, ++it)
241  {
242 
243  statismo::VectorType sampleAtPt = PixelConversionTrait<Pixel>::ToVector(it.Value());
244  for (unsigned j = 0; j < StandardImageRepresenter::GetDimensions(); j++)
245  {
246  unsigned idx = this->MapPointIdToInternalIdx(i, j);
247  sampleVec[idx] = sampleAtPt[j];
248  }
249  }
250  return sampleVec;
251 }

253 template <typename Pixel, unsigned IMAGE_DIMENSION>
254 typename StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::DatasetPointerType
255 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::SampleVectorToSample(const statismo::VectorType & sample) const
256 {
257  using DuplicatorType = itk::ImageDuplicator<DatasetType>;
258  auto duplicator = DuplicatorType::New();
259  duplicator->SetInputImage(this->m_reference);
260  duplicator->Update();
261  DatasetPointerType clonedImage = duplicator->GetOutput();
262 
263  itk::ImageRegionIterator<DatasetType> it(clonedImage, clonedImage->GetLargestPossibleRegion());
264  it.GoToBegin();
265  for (unsigned i = 0; !(it.IsAtEnd()); ++it, i++)
266  {
267 
268  statismo::VectorType valAtPoint(StandardImageRepresenter::GetDimensions());
269  for (unsigned d = 0; d < StandardImageRepresenter::GetDimensions(); d++)
270  {
271  unsigned idx = this->MapPointIdToInternalIdx(i, d);
272  valAtPoint[d] = sample[idx];
273  }
274  ValueType v = PixelConversionTrait<Pixel>::FromVector(valAtPoint);
275  it.Set(v);
276  }
277  return clonedImage;
278 }

280 template <typename Pixel, unsigned IMAGE_DIMENSION>
281 typename StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::ValueType
282 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::PointSampleFromSample(DatasetConstPointerType sample,
283  unsigned ptid) const
284 {
285  if (ptid >= GetDomain().GetNumberOfPoints())
286  {
287  throw statismo::StatisticalModelException("invalid ptid provided to PointSampleFromSample");
288  }
289 
290  // we get the point with the id from the domain, as itk does not allow us get a point via its index.
291  PointType pt = GetDomain().GetDomainPoints()[ptid];
292  typename ImageType::IndexType idx;
293  sample->TransformPhysicalPointToIndex(pt, idx);
294  ValueType value = sample->GetPixel(idx);
295  return value;
296 }

314 void
315 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::Save(const H5::Group & fg) const
316 {
317  auto origin = m_reference->GetOrigin();
318  statismo::VectorType originVec(IMAGE_DIMENSION);
319  for (unsigned i = 0; i < IMAGE_DIMENSION; i++)
320  {
321  originVec(i) = origin[i];
322  }
323  statismo::HDF5Utils::WriteVector(fg, "origin", originVec);
324 
325  auto spacing = m_reference->GetSpacing();
326  statismo::VectorType spacingVec(IMAGE_DIMENSION);
327  for (unsigned i = 0; i < IMAGE_DIMENSION; i++)
328  {
329  spacingVec(i) = spacing[i];
330  }
331  statismo::HDF5Utils::WriteVector(fg, "spacing", spacingVec);
332  statismo::GenericEigenTraits<int>::VectorType sizeVec(IMAGE_DIMENSION);
333  for (unsigned i = 0; i < IMAGE_DIMENSION; i++)
334  {
335  sizeVec(i) = m_reference->GetLargestPossibleRegion().GetSize()[i];
336  }
337  statismo::HDF5Utils::WriteVectorOfType<int>(fg, "size", sizeVec);
338 
339  auto direction = m_reference->GetDirection();
340  statismo::MatrixType directionMat(IMAGE_DIMENSION, IMAGE_DIMENSION);
341  for (unsigned i = 0; i < IMAGE_DIMENSION; i++)
342  {
343  for (unsigned j = 0; j < IMAGE_DIMENSION; j++)
344  {
345  directionMat(i, j) = direction[i][j];
346  }
347  }
348  statismo::HDF5Utils::WriteMatrix(fg, "direction", directionMat);
349  statismo::HDF5Utils::WriteInt(fg, "imageDimension", IMAGE_DIMENSION);
350 
351  H5::Group pdGroup = fg.createGroup("pointData");
352  statismo::HDF5Utils::WriteInt(pdGroup, "pixelDimension", StandardImageRepresenter::GetDimensions());
353 
354  using DoubleMatrixType = statismo::GenericEigenTraits<double>::MatrixType;
355  statismo::MatrixType pixelMat(StandardImageRepresenter::GetDimensions(), GetNumberOfPoints());
356 
357  itk::ImageRegionIterator<DatasetType> it(m_reference, m_reference->GetLargestPossibleRegion());
358  it.GoToBegin();
359  for (unsigned i = 0; !(it.IsAtEnd()); ++i, ++it)
360  {
361  pixelMat.col(i) = PixelConversionTrait<Pixel>::ToVector(it.Get());
362  }
363  DoubleMatrixType pixelMatDouble = pixelMat.cast<double>();
364  H5::DataSet ds = statismo::HDF5Utils::WriteMatrixOfType<double>(pdGroup, "pixelValues", pixelMatDouble);
365  statismo::HDF5Utils::WriteIntAttribute(ds, "datatype", PixelConversionTrait<Pixel>::GetDataType());
366 }

375 template <typename Pixel, unsigned IMAGE_DIMENSION>
376 unsigned
377 StandardImageRepresenter<Pixel, IMAGE_DIMENSION>::GetPointIdForPoint(const PointType & pt) const
378 {
379  // itks organization is slice row col
380  typename DatasetType::IndexType idx;
381  bool ptInImage = this->m_reference->TransformPhysicalPointToIndex(pt, idx);
382 
383  auto size = this->m_reference->GetLargestPossibleRegion().GetSize();
384 
385  // It does not make sense to allow points outside the image, because only the inside is modeled.
386  // However, some discretization artifacts of image and surface operations may produce points that
387  // are just on the boundary of the image, but mathematically outside. We accept these points and
388  // return the iD of the closest image point.
389  //
390  // Any points further out will trigger an exception.
391  //
392  if (!ptInImage)
393  {
394  for (unsigned int i = 0; i < ImageType::ImageDimension; ++i)
395  {
396  // As soon as one coordinate is further away than one pixel, we throw an exception.
397  if (idx[i] < -1 || idx[i] > static_cast<int>(size[i]))
398  {
399  throw statismo::StatisticalModelException(
400  "GetPointIdForPoint computed invalid ptId. Make sure that the point is within the reference you chose ");
401  }
402  // If it is on the boundary, we set it to the nearest boundary coordinate.
403  if (idx[i] == -1)
404  {
405  idx[i] = 0;
406  }
407  if (idx[i] == static_cast<int>(size[i]))
408  {
409  idx[i] = size[i] - 1;
410  }
411  }
412  }
413 
414 
415  // in itk, idx 0 is by convention the fastest moving index
416  unsigned int index = 0;
417  for (unsigned int i = 0; i < ImageType::ImageDimension; ++i)
418  {
419  unsigned int multiplier = 1;
420  for (int d = i - 1; d >= 0; --d)
421  {
422  multiplier *= size[d];
423  }
424  index += multiplier * idx[i];
425  }
426 
427  return index;
428 }
