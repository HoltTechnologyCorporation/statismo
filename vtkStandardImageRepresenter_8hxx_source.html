64 template <class TScalar, unsigned PIXEL_DIMENSIONS>
65 void
66 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::AssertCompatibility(DatasetConstPointerType data) const
67 {
68  if (sizeof(TScalar) != const_cast<vtkStructuredPoints *>(data)->GetScalarSize())
69  {
70  STATISMO_LOG_DEBUG("TScalar size: " + std::to_string(sizeof(TScalar)));
71  STATISMO_LOG_DEBUG("vtkStructuredPoints scalar size: " +
72  std::to_string(const_cast<vtkStructuredPoints *>(data)->GetScalarSize()));
73  throw StatisticalModelException("incompatible data (bad scalar type) given to representer",
74  statismo::Status::INVALID_DATA_ERROR);
75  }
76 
77  if (PIXEL_DIMENSIONS != const_cast<vtkStructuredPoints *>(data)->GetNumberOfScalarComponents())
78  {
79  throw StatisticalModelException("incompatible data (bad scalar components count) given to representer",
80  statismo::Status::INVALID_DATA_ERROR);
81  }
82 }

113 template <class TScalar, unsigned PIXEL_DIMENSIONS>
114 vtkSmartPointer<vtkStructuredPoints>
115 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::LoadRef(const H5::Group & fg) const
116 {
117 
118  auto type = HDF5Utils::ReadStringAttribute(fg, "datasetType");
119  if (type != "IMAGE")
120  {
121  throw StatisticalModelException((std::string("Cannot load representer data: The ") +
122  "representer specified in the given hdf5 file is of the wrong type: (" + type +
123  ", expected IMAGE)")
124  .c_str(),
125  Status::INVALID_H5DATA_ERROR);
126  }
127 
128  statismo::VectorType originVec;
129  HDF5Utils::ReadVector(fg, "origin", originVec);
130 
131  auto imageDimension = static_cast<unsigned>(HDF5Utils::ReadInt(fg, "imageDimension"));
132  if (imageDimension > 3)
133  {
134  throw StatisticalModelException("this representer does not support images of dimensionality > 3");
135  }
136 
137  double origin[3] = { 0, 0, 0 };
138  for (unsigned i = 0; i < imageDimension; i++)
139  {
140  origin[i] = originVec[i];
141  }
142 
143  statismo::VectorType spacingVec;
144  HDF5Utils::ReadVector(fg, "spacing", spacingVec);
145  double spacing[3] = { 0, 0, 0 };
146  for (unsigned i = 0; i < imageDimension; i++)
147  {
148  spacing[i] = spacingVec[i];
149  }
150 
151  typename statismo::GenericEigenTraits<int>::VectorType sizeVec;
152  HDF5Utils::ReadVectorOfType<int>(fg, "size", sizeVec);
153  int size[3] = { 1, 1, 1 };
154  for (unsigned i = 0; i < imageDimension; i++)
155  {
156  size[i] = sizeVec[i];
157  }
158 
159  H5::Group pdGroup = fg.openGroup("./pointData");
160 
161  typename statismo::GenericEigenTraits<double>::MatrixType pixelMat;
162  HDF5Utils::ReadMatrixOfType<double>(pdGroup, "pixelValues", pixelMat);
163 
164  auto ds = pdGroup.openDataSet("pixelValues");
165  auto datatype = static_cast<unsigned>(HDF5Utils::ReadIntAttribute(ds, "datatype"));
166 
167  if (statismo::GetDataTypeId<TScalar>() != datatype)
168  {
169  STATISMO_LOG_WARNING(
170  "The datatype specified for the scalars does not match the TPixel template argument used in the representer");
171  }
172 
173  auto newImage = vtkSmartPointer<vtkStructuredPoints>::New();
174  newImage->SetDimensions(size[0], size[1], size[2]);
175  newImage->SetExtent(0, size[0] - 1, 0, size[1] - 1, 0, size[2] - 1);
176  newImage->SetOrigin(origin[0], origin[1], origin[2]);
177  newImage->SetSpacing(spacing[0], spacing[1], spacing[2]);
178  newImage->AllocateScalars(helper::vtkStatismoDataTypeIdToVtkDataTypeId(datatype), PIXEL_DIMENSIONS);
179 
180  for (int i = 0; i < size[2]; i++)
181  {
182  for (int j = 0; j < size[1]; j++)
183  {
184  for (int k = 0; k < size[0]; k++)
185  {
186  unsigned index = size[1] * size[0] * i + size[0] * j + k;
187  auto scalarPtr = static_cast<TScalar *>(newImage->GetScalarPointer(k, j, i));
188  for (unsigned d = 0; d < PIXEL_DIMENSIONS; d++)
189  {
190  scalarPtr[d] = pixelMat(d, index);
191  }
192  }
193  }
194  }
195 
196  return newImage;
197 }

203 {
204  auto tmpfilename = statismo::utils::CreateTmpName(".vtk");
205 
206  auto uw = statismo::MakeStackUnwinder([=]() { statismo::utils::RemoveFile(tmpfilename); });
207 
208  statismo::HDF5Utils::GetFileFromHDF5(fg, "./reference", tmpfilename.c_str());
209 
210  vtkNew<vtkStructuredPointsReader> reader;
211  reader->SetFileName(tmpfilename.c_str());
212  reader->Update();
213

217  }
218 
219  return reader->GetOutput();

236 template <class TScalar, unsigned PIXEL_DIMENSIONS>
237 VectorType
238 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::SampleToSampleVector(DatasetConstPointerType sample) const
239 {
240  AssertCompatibility(sample);
241 
242  auto reference = const_cast<vtkStructuredPoints *>(this->m_reference.GetPointer());
243  auto sp = const_cast<vtkStructuredPoints *>(sample);
244 
245  if (reference->GetNumberOfPoints() != sp->GetNumberOfPoints())
246  {
247  throw StatisticalModelException("The sample does not have the correct number of points");
248  }
249 
250  VectorType sampleVector = VectorType::Zero(m_reference->GetNumberOfPoints() * PIXEL_DIMENSIONS);
251 
252  vtkDataArray * dataArray = sp->GetPointData()->GetArray(0);
253 
254  // \todo Make this more efficient using SetVoidArray of vtk
255  // \warning This is only possible, if we enforce VectorType and
256  // vtkStructuredPoints to have the same data type, e.g. float or int.
257  for (unsigned i = 0; i < m_reference->GetNumberOfPoints(); i++)
258  {
259  double val[PIXEL_DIMENSIONS];
260  dataArray->GetTuple(i, val);
261  for (unsigned d = 0; d < PIXEL_DIMENSIONS; d++)
262  {
263  sampleVector(vtkStandardImageRepresenter::MapPointIdToInternalIdx(i, d)) = val[d];
264  }
265  }
266 
267  return sampleVector;
268 }

271 typename vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::DatasetPointerType
272 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::SampleVectorToSample(const VectorType & sampleVector) const
273 {
274  auto sample = vtkSmartPointer<vtkStructuredPoints>::New();
275  auto reference = const_cast<vtkStructuredPoints *>(m_reference.GetPointer());
276  sample->DeepCopy(reference);
277 
278  vtkDataArray * dataArray = sample->GetPointData()->GetArray(0);
279  for (unsigned i = 0; i < GetNumberOfPoints(); i++)
280  {
281  double val[PIXEL_DIMENSIONS];
282  for (unsigned d = 0; d < PIXEL_DIMENSIONS; d++)
283  {
284  val[d] = sampleVector(vtkStandardImageRepresenter::MapPointIdToInternalIdx(i, d));
285  }
286  dataArray->SetTuple(i, val);
287  }
288 
289  return sample;
290 }

292 template <class TScalar, unsigned PIXEL_DIMENSIONS>
293 typename vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::ValueType
294 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::PointSampleFromSample(DatasetConstPointerType sample,
295  unsigned ptid) const
296 {
297  AssertCompatibility(sample);
298 
299  auto mutableSample = const_cast<vtkStructuredPoints *>(sample);
300  if (ptid >= mutableSample->GetNumberOfPoints())
301  {
302  throw StatisticalModelException("invalid ptid provided to PointSampleFromSample");
303  }
304 
305  double doubleVal[PIXEL_DIMENSIONS];
306  mutableSample->GetPointData()->GetScalars()->GetTuple(ptid, doubleVal);
307  ValueType val(PIXEL_DIMENSIONS);
308 
309  // vtk returns double. We need to convert it to whatever precision we are using in NDPixel
310  for (unsigned i = 0; i < PIXEL_DIMENSIONS; i++)
311  {
312  val[i] = static_cast<TScalar>(doubleVal[i]);
313  }
314  return val;
315 }

343 template <class TScalar, unsigned PIXEL_DIMENSIONS>
344 void
345 vtkStandardImageRepresenter<TScalar, PIXEL_DIMENSIONS>::Save(const H5::Group & fg) const
346 {
347  using namespace H5;
348 
349  // get the effective image dimension, by check the size
350  int * size = m_reference->GetDimensions();
351  unsigned imageDimension = 0;
352  if (size[2] == 1 && size[1] == 1)
353  {
354  imageDimension = 1;
355  }
356  else if (size[2] == 1)
357  {
358  imageDimension = 2;
359  }
360  else
361  {
362  imageDimension = 3;
363  }
364 
365  HDF5Utils::WriteInt(fg, "imageDimension", imageDimension);
366 
367  double * origin = m_reference->GetOrigin();
368  statismo::VectorType originVec = statismo::VectorType::Zero(imageDimension);
369  for (unsigned i = 0; i < imageDimension; i++)
370  {
371  originVec(i) = origin[i];
372  }
373  HDF5Utils::WriteVector(fg, "origin", originVec);
374 
375  double * spacing = m_reference->GetSpacing();
376  statismo::VectorType spacingVec = statismo::VectorType::Zero(imageDimension);
377  for (unsigned i = 0; i < imageDimension; i++)
378  {
379  spacingVec(i) = spacing[i];
380  }
381  HDF5Utils::WriteVector(fg, "spacing", spacingVec);
382 
383  statismo::GenericEigenTraits<int>::VectorType sizeVec =
384  statismo::GenericEigenTraits<int>::VectorType::Zero(imageDimension);
385  for (unsigned i = 0; i < imageDimension; i++)
386  {
387  sizeVec(i) = size[i];
388  }
389  HDF5Utils::WriteVectorOfType<int>(fg, "size", sizeVec);
390 
391  // VTK does not support image direction. We write the identity matrix
392  statismo::MatrixType directionMat = statismo::MatrixType::Identity(imageDimension, imageDimension);
393  HDF5Utils::WriteMatrix(fg, "direction", directionMat);
394 
395  using DoubleMatrixType = statismo::GenericEigenTraits<double>::MatrixType;
396  DoubleMatrixType pixelMat(vtkStandardImageRepresenter::GetDimensions(), GetNumberOfPoints());
397 
398  for (unsigned i = 0; i < static_cast<unsigned>(size[2]); i++)
399  {
400  for (unsigned j = 0; j < static_cast<unsigned>(size[1]); j++)
401  {
402  for (unsigned k = 0; k < static_cast<unsigned>(size[0]); k++)
403  {
404  unsigned ind = i * size[1] * size[0] + j * size[0] + k;
405  auto pixel = static_cast<TScalar *>(m_reference->GetScalarPointer(k, j, i));
406  for (unsigned d = 0; d < vtkStandardImageRepresenter::GetDimensions(); d++)
407  {
408  pixelMat(d, ind) = pixel[d];
409  }
410  }
411  }
412  }
413 
414  auto pdGroup = fg.createGroup("pointData");
415  statismo::HDF5Utils::WriteInt(pdGroup, "pixelDimension", vtkStandardImageRepresenter::GetDimensions());
416 
417  auto ds = HDF5Utils::WriteMatrixOfType<double>(pdGroup, "pixelValues", pixelMat);
418  HDF5Utils::WriteIntAttribute(ds, "datatype", statismo::GetDataTypeId<TScalar>());
419 }
