51  MatrixType
52  ComputeEigenfunctionsAtPoint(const PointType & pt) const
53  {
54  unsigned kernelDim = m_kernel.GetDimension();
55 
56  // for every domain point x in the list, we compute the kernel vector
57  // kx = (k(x, x1), ... k(x, xm))
58  // since the kernel is matrix valued, kx is actually a matrix
59  MatrixType kxi = MatrixType::Zero(kernelDim, m_nystromPoints.size() * kernelDim);
60 
61  for (unsigned j = 0; j < m_nystromPoints.size(); j++)
62  {
63  kxi.block(0, j * kernelDim, kernelDim, kernelDim) = m_kernel(pt, m_nystromPoints[j]);
64  }
65 
66 
67  MatrixType resMat = MatrixType::Zero(kernelDim, m_numEigenfunctions);
68  for (unsigned j = 0; j < m_numEigenfunctions; j++)
69  {
70  MatrixType x = (kxi * m_nystromMatrix.col(j));
71  resMat.block(0, j, kernelDim, 1) = x;
72  }
73  return resMat;
74  }

88 private:
89  Nystrom(const Representer<T> * representer,
90  const MatrixValuedKernel<PointType> & kernel,
91  unsigned numEigenfunctions,
92  unsigned numberOfPointsForApproximation)
93  : m_representer(representer)
94  , m_kernel(kernel)
95  , m_numEigenfunctions(numEigenfunctions)
96  {
97  DomainType domain = m_representer->GetDomain();
98  m_nystromPoints = GetNystromPoints(domain, numberOfPointsForApproximation);
99  auto numDomainPoints = domain.GetNumberOfPoints();
100 
101  // compute a eigenvalue decomposition of the kernel matrix, evaluated at the points used for the
102  // nystrom approximation
103 
104  MatrixType U; // will hold the eigenvectors (principal components)
105  VectorType D; // will hold the eigenvalues (variance)
106  ComputeKernelMatrixDecomposition(&m_kernel, m_nystromPoints, numEigenfunctions, U, D);
107 
108  // precompute the part of the nystrom approximation, which is independent of the domain point
109  float normFactor = static_cast<float>(m_nystromPoints.size()) / static_cast<float>(numDomainPoints);
110  m_nystromMatrix =
111  std::sqrt(normFactor) * (U.leftCols(numEigenfunctions) * D.topRows(numEigenfunctions).asDiagonal().inverse());
112 
113  m_eigenvalues = (1.0f / normFactor) * D.topRows(numEigenfunctions);
114  }

139  void
140  ComputeKernelMatrixDecomposition(const MatrixValuedKernel<PointType> * kernel,
141  const std::vector<PointType> & xs,
142  unsigned numComponents,
143  MatrixType & U,
144  VectorType & D) const
145  {
146  unsigned kernelDim = kernel->GetDimension();
147 
148  auto n = xs.size();
149  MatrixTypeDoublePrecision K = MatrixTypeDoublePrecision::Zero(n * kernelDim, n * kernelDim);
150  for (unsigned i = 0; i < n; ++i)
151  {
152  for (unsigned j = i; j < n; ++j)
153  {
154  MatrixType k_xixj = (*kernel)(xs[i], xs[j]);
155  for (unsigned d1 = 0; d1 < kernelDim; d1++)
156  {
157  for (unsigned d2 = 0; d2 < kernelDim; d2++)
158  {
159  double elem_d1d2 = k_xixj(d1, d2);
160  K(i * kernelDim + d1, j * kernelDim + d2) = elem_d1d2;
161  K(j * kernelDim + d2, i * kernelDim + d1) = elem_d1d2;
162  }
163  }
164  }
165  }
166 
167  using SVDType = RandSVD<double>;
168  SVDType svd(K, numComponents * kernelDim);
169  U = svd.MatrixU().cast<ScalarType>();
170  D = svd.SingularValues().cast<ScalarType>();
171  }
