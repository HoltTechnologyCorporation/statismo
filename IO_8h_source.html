106  static UniquePtrType<StatisticalModelType>
107  LoadStatisticalModel(typename StatisticalModelType::RepresenterType * representer,
108  const H5::Group & modelRoot,
109  unsigned maxNumberOfPCAComponents = std::numeric_limits<unsigned>::max())
110  {
111 
112  UniquePtrType<StatisticalModelType> newModel;
113  ModelInfo modelInfo;
114 
115  try
116  {
117  representer->Load(modelRoot.openGroup("./representer"));
118 
119  int minorVersion = 0;
120  int majorVersion = 0;
121 
122  if (!HDF5Utils::ExistsObjectWithName(modelRoot, "version"))
123  {
124  // This is an old statismo format, that was versioned.
125  // We set the version to 0.8 as this is the last
126  // version that stores the old format.
127  if (representer->GetLogger())
128  {
129  representer->GetLogger()->Log(LogEntry{ "version attribute does not exist in hdf5 file. Assuming version 0.8",
130  __FILE__,
131  std::to_string(__LINE__) },
132  LogLevel::LOG_WARNING);
133  }
134 
135  minorVersion = gk_oldFileVersionMinor;
136  majorVersion = gk_oldFileVersionMajor;
137  }
138  else
139  {
140  auto versionGroup = modelRoot.openGroup("./version");
141  minorVersion = HDF5Utils::ReadInt(versionGroup, "./minorVersion");
142  majorVersion = HDF5Utils::ReadInt(versionGroup, "./majorVersion");
143  }
144 
145  auto modelGroup = modelRoot.openGroup("./model");
146  VectorType mean;
147  HDF5Utils::ReadVector(modelGroup, "./mean", mean);
148  VectorType pcaVariance;
149  HDF5Utils::ReadVector(modelGroup, "./pcaVariance", maxNumberOfPCAComponents, pcaVariance);
150 
151  auto noiseVariance = HDF5Utils::ReadFloat(modelGroup, "./noiseVariance");
152 
153  // Depending on the statismo version, the pcaBasis matrix was stored as U*D or U (where U are the orthonormal PCA
154  // Basis functions and D the standard deviations). Here we make sure that we fill the pcaBasisMatrix (which
155  // statismo stores as U*D) with the right values.
156  MatrixType pcaBasisMatrix;
157  if (majorVersion == gk_oldFileVersionMinor && minorVersion == gk_oldFileVersionMajor)
158  {
159  HDF5Utils::ReadMatrix(modelGroup, "./pcaBasis", maxNumberOfPCAComponents, pcaBasisMatrix);
160  VectorType D = pcaVariance.array().sqrt(); // NOLINT
161  MatrixType orthonormalPCABasisMatrix = pcaBasisMatrix * DiagMatrixType(D).inverse();
162  newModel =
163  StatisticalModelType::SafeCreate(representer, mean, orthonormalPCABasisMatrix, pcaVariance, noiseVariance);
164  }
165  else if (majorVersion == gk_currenFileVersionMajor && minorVersion == gk_currentFileVersionMinor)
166  {
167  HDF5Utils::ReadMatrix(modelGroup, "./pcaBasis", maxNumberOfPCAComponents, pcaBasisMatrix);
168  newModel = StatisticalModelType::SafeCreate(representer, mean, pcaBasisMatrix, pcaVariance, noiseVariance);
169  }
170  else
171  {
172  std::ostringstream os;
173  os << "an invalid statismo version was provided (" << majorVersion << "." << minorVersion << ")";
174  throw StatisticalModelException(os.str().c_str(), Status::BAD_VERSION_ERROR);
175  }
176 
177  modelInfo.Load(modelRoot);
178  }
179  catch (H5::Exception & e)
180  {
181  std::string msg(std::string("an exeption occured while reading HDF5 file") +
182  "The most likely cause is that the hdf5 file does not contain the required objects. \n" +
183  e.getCDetailMsg());
184  throw StatisticalModelException(msg.c_str(), Status::INVALID_DATA_ERROR);
185  }
186 
187  newModel->SetModelInfo(modelInfo);
188  return newModel;
189  }

244  static void
245  SaveStatisticalModel(const StatisticalModelType & model, const H5::Group & modelRoot)
246  {
247  try
248  {
249  // create the group structure
250  using RepresenterType = typename StatisticalModelType::RepresenterType;
251  auto dataTypeStr = RepresenterType::TypeToString(model.GetRepresenter()->GetType());
252  auto representerGroup = modelRoot.createGroup("./representer");
253 
254  HDF5Utils::WriteStringAttribute(representerGroup, "name", model.GetRepresenter()->GetName());
255  HDF5Utils::WriteStringAttribute(representerGroup, "version", model.GetRepresenter()->GetVersion());
256  HDF5Utils::WriteStringAttribute(representerGroup, "datasetType", dataTypeStr);
257 
258  model.GetRepresenter()->Save(representerGroup);
259 
260  auto modelGroup = modelRoot.createGroup("./model");
261  HDF5Utils::WriteMatrix(modelGroup, "./pcaBasis", model.GetOrthonormalPCABasisMatrix());
262  HDF5Utils::WriteVector(modelGroup, "./pcaVariance", model.GetPCAVarianceVector());
263  HDF5Utils::WriteVector(modelGroup, "./mean", model.GetMeanVector());
264  HDF5Utils::WriteFloat(modelGroup, "./noiseVariance", model.GetNoiseVariance());
265 
266  model.GetModelInfo().Save(modelRoot);
267  }
268  catch (H5::Exception & e)
269  {
270  std::string msg(std::string("an exception occurred while writing HDF5 file \n") + e.getCDetailMsg());
271  throw StatisticalModelException(msg.c_str(), Status::IO_ERROR);
272  }
273  }
