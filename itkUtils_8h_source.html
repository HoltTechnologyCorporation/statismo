76 inline std::vector<std::string>
77 GetDirFiles(const std::string & dir, const std::string & extension = ".*")
78 {
79  auto directory = ::itk::Directory::New();
80  directory->Load(dir.c_str());
81 
82  std::vector<std::string> files;
83  for (unsigned i = 0; i < directory->GetNumberOfFiles(); i++)
84  {
85  const char * filename = directory->GetFile(i);
86  if (extension == ".*" || std::string(filename).find(extension) != std::string::npos)
87  {
88  files.emplace_back(filename);
89  }
90  }
91 
92  return files;
93 }

99 template <typename DataType>
100 typename DataType::Pointer
101 CloneMesh(typename DataType::Pointer pMesh)
102 {
103  using IdentityTransformType = ::itk::IdentityTransform<float, DataType::PointDimension>;
104  using TransformFilterType = ::itk::TransformMeshFilter<DataType, DataType, IdentityTransformType>;
105  auto transformMeshFilter = TransformFilterType::New();
106  auto identityTransform = IdentityTransformType::New();
107  transformMeshFilter->SetInput(pMesh);
108  transformMeshFilter->SetTransform(identityTransform);
109  transformMeshFilter->Update();
110 
111  return transformMeshFilter->GetOutput();
112 }
